Buttons
  *workBtn
    name:Perform Work
    desc:Use this button to manually do work for your tribe.
    on click:yield random(0, 2) water
    on click:yield random(0, 2) fruit
    on click:yield random(0, 2) meat
    on click:yield random(0, 2) wood
    on click:yield random(0, 2) stone
    on click:anim icon wobble
    class:bigButton hasFlares
  *gainFertilizerBtn
    name:Buy Fertilizer
    desc:Click to buy fertilizer(cost 5 water, 5 fruit). Each fertilizer makes farmers <b>twice as efficient.</b>
    on click:yield 1 fertilizer
    on click:yield -5 water
    on click:yield -5 fruit
  *showAvailUpgr
    name:Available Upgrades
    tag:upgradeSwitcher
    class:upgradeSwitch
    on click:
      show :Upgrades:notOwned
      hide :Upgrades:owned
      anim glow
    end
    no tooltip
  *showOwnedUpgr
    name:Owned Upgrades
    tag:upgradeSwitcher
    class:upgradeSwitch
    on click:
      hide :Upgrades:notOwned
      show :Upgrades:owned
      anim glow
    end
    no tooltip

Buildings
  *backgroundWorker
    start with:1
    always hidden
    on tick:yield 0.1 people
    on tick:yield (-3 * totalPopulation) water
    on tick:yield (-3 * totalPopulation) fruit
    on tick:yield (-3 * totalPopulation) meat
    on tick:
      show gainFertilizerBtn
      if (ItemsLeft = 0) hide gainFertilizerBtn
      if (water < 5) hide gainFertilizerBtn
      if (fruit < 5) hide gainFertilizerBtn
      if (farmers = 0) hide gainFertilizerBtn
    end
  *well|wells
    name:Well|Wells
    desc:A source of water
    cost:5 stone,3 wood
    on tick:yield 5 water
  *hunter|hunters
    name:Hunter|Hunters
    cost refund:100%
    cost increase:100%
    cost:1 people
    desc:Hunters are people in the tribe who bring back meat.
    on tick:yield 10 meat
  *farmer|farmers
    name:Farmer|Farmers
    cost refund:100%
    cost increase:100%
    cost:1 people,5 seeds
    desc:Farmers grow fruits for your tribe.
    on tick:yield 10 fruit
    on tick:yield 10 seeds
  *lumberjack|lumberjacks
    name:Lumberjack|Lumberjacks
    cost refund:100%
    cost increase:100%
    cost:1 people
    desc:Lumberbacks cut down wood for your tribe.
    on tick:yield 10 wood
  *miner|miners
    name:Miner|Miners
    cost refund:100%
    cost increase:100%
    cost:1 people
    desc:Miners mine stone for your tribe.
